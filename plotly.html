Plot that shows the number of items ordered in each department.

instacart_department =
  count(instacart, department) %>% 
  arrange(n) %>% 
  mutate(
    department = factor(department, levels = department)
  )


instacart_department %>% 
  plot_ly(x = ~department, y = ~n, color = ~department, type = "bar", colors = "viridis")

Import NYC Restaurant Inspections data.

data("rest_inspec")
rest_inspec %>% 
  filter(grade %in% c("A", "B", "C"), boro != "Missing") %>% 
  mutate(boro = str_to_title(boro)) %>% 
  filter(str_detect(dba, regex("pizza", ignore_case = TRUE))) %>%
  mutate(
    boro = fct_infreq(boro),
    boro = fct_recode(boro, "The City" = "Manhattan")) %>% 
  plot_ly(
    x = ~boro, y = ~score, color = ~boro, 
    type = "box", colors = "viridis"
